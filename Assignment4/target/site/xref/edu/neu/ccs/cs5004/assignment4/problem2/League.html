3   /**
4    * Represents a class League with their details -name, startMonth, endMonth, gameNumber, nextGame.
5    *
6    * @author nikkiwang
7    */
8   public class League implements ILeague {
9   
10    private String name;
11    private int startMonth;
12    private int endMonth;
13    private int gameNumber;
14    private Game nextGame;
15    private Game thisGame;
16  
17    /**
18     * Creat a league object.
19     *
20     * @param name -- league's name
21     * @param startMonth -- league's start month
22     * @param endMonth -- league's end month
23     * @param gameNumber -- league's game number
24     * @throws Exception if the month is not valid
25     */
26    public League(String name, int startMonth, int endMonth, int gameNumber)
27        throws Exception {
28      if (startMonth < 1 || startMonth > 12 || endMonth < 1 || endMonth > 12
29          || startMonth > endMonth) {
30        throw new IllegalArgumentException("invalid month.");
31      }
32      this.name = name;
33      this.startMonth = startMonth;
34      this.endMonth = endMonth;
35      this.gameNumber = gameNumber;
36      this.nextGame = null;
37      this.thisGame = null;
38    }
39  
40    /**
41     * get league name.
42     *
43     * @return league name
44     */
45    public String getName() {
46      return this.name;
47    }
48

50     * Set next game.
51     */
52    @Override

55    }

76  
77    /**
78     * Compare Team.
79     *
80     * @return the winning team.
81     */
82    @Override
83    public Team compareTeam(Team team1, Team team2) throws IllegalArgumentException {
84      if (team1.getLeague() != team2.getLeague()) {
85        throw new IllegalArgumentException("Not in the same league.");
86      }
87      int team1Point = team1.getPoints();
88      int team2Point = team2.getPoints();
89      if (team1Point == team2Point) {
90        int team1WinNum = team1.getRecord().getWinNum();
91        int team2WinNum = team2.getRecord().getWinNum();
92        if (team1WinNum == team2WinNum) {
93          throw new TiedTeamsException("Two teams are tied.");
94        }
95        return (team1WinNum > team2WinNum) ? team1 : team2;
96      }
97      return (team1Point > team2Point) ? team1 : team2;
98    }
99  
100   /**
101    * Schedule the next Game.
102    */
103   @Override
104   public void scheduleGame(Game game, boolean seasonOn) throws IllegalArgumentException {
105     this.setNextGame(game);
106     this.nextGame.getHomeTeam().setSeasonOn(seasonOn);
107     this.nextGame.getAwayTeam().setSeasonOn(seasonOn);
108     this.nextGame.getAwayTeam().setGameRemaining(this.gameNumber);
109     this.nextGame.getHomeTeam().setGameRemaining(this.gameNumber);
110   }
111 
112   /**
113    * Play the game, update all information.
114    *
115    * @param game -- game
116    * @param homeTeamScore -- home team score
117    * @param awayTeamScore -- away team score
118    * @throws IllegalArgumentException if date is not within season
119    */
120   @Override
121   public void playGame(Game game, int homeTeamScore, int awayTeamScore)
122       throws IllegalArgumentException {
123     if (game.getDate().getMonth() < this.startMonth || game.getDate().getMonth() > this.endMonth) {
124       throw
125           new IllegalArgumentException("invalid game date");
126     }
127     this.thisGame = game;
128     this.thisGame.setPlayed(true, homeTeamScore, awayTeamScore);
129     if (homeTeamScore > awayTeamScore) {
130       this.thisGame.getHomeTeam().getRecord().addWinNum();
131       this.thisGame.getAwayTeam().getRecord().addLossNum();
132     } else if (homeTeamScore == awayTeamScore) {
133       this.thisGame.getHomeTeam().getRecord().addTieNum();
134       this.thisGame.getAwayTeam().getRecord().addTieNum();
135     } else {
136       this.thisGame.getAwayTeam().getRecord().addWinNum();
137       this.thisGame.getHomeTeam().getRecord().addLossNum();
138     }
139   }
