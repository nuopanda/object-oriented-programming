3   /**
4    * Represents a class BaseballTeam as a subclass of Team with their details -name, league, seasonOn,
5    * gamePlayed, gameRemaining,record, points, lastGame and WINPOINT.
6    *
7    * @author nikkiwang
8    */
9   public class BaseballTeam extends Team {
10  
11    /**
12     * Create an BaseballTeam object.
13     *
14     * @param name -- team's name
15     * @param league -- team's league
16     */
17    public BaseballTeam(String name, League league) {
18      super(name, league);
19      this.record.setTieNum(null);
20    }
21  
22    /**
23     * Play game, update information. update lastGame, gamePlayed, gameRemaining, record
24     *
25     * @param game --played game
26     * @param homeTeam -- whether the team is homeTeam
27     */
28    @Override
29    public void playGame(Game game, boolean homeTeam) {
30      this.lastGame = game;
31      this.gamePlayed += 1;
32      this.gameRemaining -= 1;
33      int homeTeamScore = game.getHomeTeamScore();
34      int awayTeamScore = game.getAwayTeamScore();
35      if (homeTeam) {
36        if (homeTeamScore > awayTeamScore) {
37          this.getRecord().addWinNum();
38        } else if (homeTeamScore < awayTeamScore) {
39          this.getRecord().addLossNum();
40        }
41      } else {
42        if (homeTeamScore < awayTeamScore) {
43          this.getRecord().addWinNum();
44        } else if (homeTeamScore > awayTeamScore) {
45          this.getRecord().addLossNum();
46        }
47      }
48      this.getPoints();
49    }
50  }
