5   public class AutoSystem {
6     private static final String OUTPUT_DIR = "--output-dir";
7     private static final String CSV_FILE = "--csv-file";
8     private static final String EMAIL_TEMPLATE = "--email-template";
9     private static final String LETTER_TEMPLATE = "--letter-template";
10  
11  
12    protected static void main(String[] args) {
13  
14      Options options = new Options();
15      options.addOptions(new Option("--email", "only generate email messages", false));
16      options.addOptions(new Option("--email-template", "accept a filename that holds the template. "
17          + "Required if --email is used", true));
18      options.addOptions(new Option("--letter", "only generate letters", false));
19      options.addOptions(new Option("--letter-template", "accept a filename that hold the "
20          + "letter template. Required if --letter is used", true));
21      options.addOptions(new Option("--output-dir", "accept the name of a folder, all\n"
22          + "output is placed in this folder", true));
23      options.addOptions(new Option("--csv-file", " accept the name of the csv file to\n"
24          + "process", true));
25  
26      /*String[] test = new String[]{
27          "--email", "--email-template", "email-template.txt",
28          "--output-dir", "emails", "--csv-file", "customer.csv"};
29      */
30      String templateName = null;
31      String outputDir = null;
32      String csvFile = null;
33  
34      // determine the template and csvFile to read.
35      CmdParser cmdParser = new CmdParser();
36      CmdLineResult cmdLineResult = cmdParser.parse(args, options);
37      for (Map.Entry<String, Option> entry : cmdLineResult.getOptions().entrySet()) {
38        if (entry.getKey().equals(EMAIL_TEMPLATE) && entry.getValue().getArgs() != null) {
39          templateName = entry.getValue().getArgs();
40        }
41        if (entry.getKey().equals(LETTER_TEMPLATE) && entry.getValue().getArgs() != null) {
42          templateName = entry.getValue().getArgs();
43        }
44        if (entry.getKey().equals(OUTPUT_DIR)) {
45          outputDir = entry.getValue().getArgs();
46        }
47        if (entry.getKey().equals(CSV_FILE)) {
48          csvFile = entry.getValue().getArgs();
49        }
50  
51      }
52  
53  
54      String template = TemplateParser.parse(templateName);
55      CustomerInfo customerInfo = CsvParser.parse(csvFile);
56      // output dir should be provided by the user
57      FileGenerator.generator(customerInfo, template, outputDir);
58    }
